20/// @brief Deduce the MPI_Datatype to use as send_type in a p2p send operation.If \ref kamping::send_type() is given,
21/// the \c MPI_Datatype wrapped inside will be used as send_type. Otherwise, the \c MPI_datatype is derived
22/// automatically based on send_buf's underlying \c value_type.
23///
24/// @tparam send_value_type Value type of the send buffer.
25/// @tparam Args Types of all arguments passed to the wrapped MPI call.
26/// @param args All arguments passed to a wrapped MPI call.
27/// @return Return the \c MPI send_type wrapped in a DataBuffer. This is either an lvalue reference to the
28/// send_type DataBuffer if the send_type is provided by the user or a newly created send_type DataBuffer
29/// otherwise.
30template <typename send_value_type, typename... Args>
31constexpr auto determine_mpi_send_datatype(Args&... args)
32 -> decltype(internal::select_parameter_type_or_default<
33 internal::ParameterType::send_type,
34 decltype(kamping::send_type_out())>(std::make_tuple(), args...)
35 .construct_buffer_or_rebind()) {
36 // Some assertions:
37 // If a send_type is given, the send_count information has to be provided, too.
38 constexpr bool is_send_type_given_as_in_param = is_parameter_given_as_in_buffer<ParameterType::send_type, Args...>;
39 if constexpr (is_send_type_given_as_in_param) {
40 constexpr bool is_send_count_given = is_parameter_given_as_in_buffer<ParameterType::send_count, Args...>;
41 static_assert(
42 is_send_count_given,
43 "If a custom send type is provided, the send count has to be provided, too."
44 );
45 }
46 // Get the send type
47 using default_mpi_send_type = decltype(kamping::send_type_out());
48
49 auto mpi_send_type =
50 internal::select_parameter_type_or_default<internal::ParameterType::send_type, default_mpi_send_type>(
51 std::make_tuple(),
52 args...
53 )
54 .construct_buffer_or_rebind();
55
56 // assure that our expectation about the return value value category (lvalue or pr-value) is true. This ensures
57 // that the return value of the function does not become a dangling rvalue reference bound to a function-local
58 // object.
59 static_assert(
60 !std::is_rvalue_reference_v<decltype(mpi_send_type)>,
61 "mpi_send_type is either a lvalue reference (in this case it returned by reference), or a non-reference type "
62 "(in "
63 "this case it is returned by value)."
64 );
65
66 if constexpr (!is_send_type_given_as_in_param) {
67 mpi_send_type.underlying() = mpi_datatype<send_value_type>();
68 }
69
70 return mpi_send_type;
71}

73/// @brief Deduce the MPI_Datatype to use as recv_type in a p2p recv operation.If \ref kamping::recv_type() is given,
74/// the \c MPI_Datatype wrapped inside will be used as recv_type. Otherwise, the \c MPI_datatype is derived
75/// automatically based on recv_buf's underlying \c value_type.
76///
77/// @tparam recv_value_type Value type of the recv buffer.
78/// @tparam recv_buf Type of the recv buffer.
79/// @tparam Args Types of all arguments passed to the wrapped MPI call.
80/// @param args All arguments passed to a wrapped MPI call.
81/// @return Return the \c MPI recv_type wrapped in a DataBuffer. This is either an lvalue reference to the
82/// recv_type DataBuffer if the recv_type is provided by the user or a newly created recv_type DataBuffer
83/// otherwise.
84template <typename recv_value_type, typename recv_buf, typename... Args>
85constexpr auto determine_mpi_recv_datatype(Args&... args)
86 -> decltype(internal::select_parameter_type_or_default<
87 internal::ParameterType::recv_type,
88 decltype(kamping::recv_type_out())>(std::make_tuple(), args...)
89 .construct_buffer_or_rebind()) {
90 // Get the recv type
91 using default_mpi_recv_type = decltype(kamping::recv_type_out());
92
93 auto mpi_recv_type =
94 internal::select_parameter_type_or_default<internal::ParameterType::recv_type, default_mpi_recv_type>(
95 std::make_tuple(),
96 args...
97 )
98 .construct_buffer_or_rebind();
99
100 // assure that our expectation about the return value value category (lvalue or pr-value) is true. This ensures
101 // that the return value of the function does not become a dangling rvalue reference bound to a function-local
102 // object.
103 static_assert(
104 !std::is_rvalue_reference_v<decltype(mpi_recv_type)>,
105 "mpi_recv_type is either a lvalue reference (in this case it returned by reference), or a non-reference type "
106 "(in "
107 "this case it is returned by value)."
108 );
109
110 constexpr bool is_recv_type_given_as_in_param = is_parameter_given_as_in_buffer<ParameterType::recv_type, Args...>;
111 // Recv buffer resize policy assertion
112 constexpr bool do_not_resize_recv_buf = std::remove_reference_t<recv_buf>::resize_policy == no_resize;
113 static_assert(
114 !is_recv_type_given_as_in_param || do_not_resize_recv_buf,
115 "If a custom recv type is given, kamping is not able to deduce the correct size of the "
116 "recv buffer. "
117 "Therefore, a sufficiently large recv buffer (with resize policy \"no_resize\") must be provided by "
118 "the user."
119 );
120
121 if constexpr (!is_recv_type_given_as_in_param) {
122 mpi_recv_type.underlying() = mpi_datatype<recv_value_type>();
123 }
124
125 return mpi_recv_type;
126}
