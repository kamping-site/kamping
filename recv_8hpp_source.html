40/// @brief Wrapper for \c MPI_Recv.
41///
42/// This wraps \c MPI_Recv. This operation performs a standard blocking receive.
43/// If the \ref kamping::recv_counts() parameter is not specified, this first performs a probe, followed by a receive of
44/// the probed message with the probed message size.
45///
46/// The following parameter is optional, but leads to an additional call to \c MPI_Probe if not present:
47/// - \ref kamping::recv_count() the number of elements to receive. Will be probed before receiving if not given.
48///
49/// The following parameters are optional:
50/// - \ref kamping::recv_buf() the buffer to receive the message into. The buffer's underlying
51/// storage must be large enough to hold all received elements. If no \ref kamping::recv_buf() is provided, the \c
52/// value_type of the recv buffer has to be passed as a template parameter to \c recv().
53///
54/// - \ref kamping::recv_type() specifying the \c MPI datatype to use as recv type. If omitted, the \c MPI datatype is
55/// derived automatically based on recv_buf's underlying \c value_type.
56///
57/// - \ref kamping::source() receive a message sent from this source rank. Defaults to probing for an arbitrary source,
58/// i.e. \c source(rank::any).
59///
60/// - \ref kamping::tag() recv message with this tag. Defaults to receiving for an arbitrary tag, i.e. \c
61/// tag(tags::any).
62///
63/// - \c kamping::status(ignore<>) or \ref kamping::status_out(). Returns info about the received message by setting the
64/// appropriate fields in the status object passed by the user. If \ref kamping::status_out() is passed, constructs a
65/// status object which may be retrieved by the user. The status can be ignored by passing \c
66/// kamping::status(kamping::ignore<>). This is the default.
67///
68/// @tparam Args Automatically deduced template parameters.
69/// @param args All required and any number of the optional parameters described above.
70/// @return Result object wrapping the output parameters to be returned by value.
71///
72/// @see \ref docs/parameter_handling.md for general information about parameter handling in KaMPIng.
73/// <hr>
74/// \include{doc} docs/resize_policy.dox
75template <
76 template <typename...>
77 typename DefaultContainerType,
78 template <typename, template <typename...> typename>
79 typename... Plugins>
80template <typename recv_value_type_tparam /* = kamping::internal::unused_tparam */, typename... Args>
81auto kamping::Communicator<DefaultContainerType, Plugins...>::recv(Args... args) const {
82 using namespace kamping::internal;
83 KAMPING_CHECK_PARAMETERS(
84 Args,
85 KAMPING_REQUIRED_PARAMETERS(),
86 KAMPING_OPTIONAL_PARAMETERS(recv_buf, tag, source, recv_count, recv_type, status)
87 );
88 using default_recv_buf_type = decltype(kamping::recv_buf(alloc_new<DefaultContainerType<recv_value_type_tparam>>));
89 auto recv_buf =
90 internal::select_parameter_type_or_default<internal::ParameterType::recv_buf, default_recv_buf_type>(
91 std::tuple<>(),
92 args...
93 )
94 .template construct_buffer_or_rebind<DefaultContainerType, internal::serialization_support_tag>();
95 constexpr bool is_serialization_used = internal::buffer_uses_serialization<decltype(recv_buf)>;
96 if constexpr (is_serialization_used) {
97 KAMPING_UNSUPPORTED_PARAMETER(Args, recv_count, when using serialization);
98 KAMPING_UNSUPPORTED_PARAMETER(Args, recv_type, when using serialization);
99 }
100 using recv_value_type = typename std::remove_reference_t<decltype(recv_buf)>::value_type;
101 static_assert(
102 !std::is_same_v<recv_value_type, internal::unused_tparam>,
103 "No recv_buf parameter provided and no receive value given as template parameter. One of these is required."
104 );
105
106 auto recv_type = internal::determine_mpi_recv_datatype<recv_value_type, decltype(recv_buf)>(args...);
107 [[maybe_unused]] constexpr bool recv_type_is_in_param = !internal::has_to_be_computed<decltype(recv_type)>;
108
109 using default_source_buf_type = decltype(kamping::source(rank::any));
110
111 auto&& source_param =
112 internal::select_parameter_type_or_default<internal::ParameterType::source, default_source_buf_type>(
113 {},
114 args...
115 );
116
117 using default_tag_buf_type = decltype(kamping::tag(tags::any));
118
119 auto&& tag_param =
120 internal::select_parameter_type_or_default<internal::ParameterType::tag, default_tag_buf_type>({}, args...);
121
122 constexpr auto tag_type = std::remove_reference_t<decltype(tag_param)>::tag_type;
123 if constexpr (tag_type == internal::TagType::value) {
124 int tag = tag_param.tag();
125 KAMPING_ASSERT(
126 Environment<>::is_valid_tag(tag),
127 "invalid tag " << tag << ", must be in range [0, " << Environment<>::tag_upper_bound() << "]"
128 );
129 }
130
131 using default_status_param_type = decltype(kamping::status(kamping::ignore<>));
132
133 auto status =
134 internal::select_parameter_type_or_default<internal::ParameterType::status, default_status_param_type>(
135 {},
136 args...
137 )
138 .construct_buffer_or_rebind();
139
140 // Get the optional recv_count parameter. If the parameter is not given,
141 // allocate a new container.
142 using default_recv_count_type = decltype(kamping::recv_count_out());
143 auto recv_count_param =
144 internal::select_parameter_type_or_default<internal::ParameterType::recv_count, default_recv_count_type>(
145 std::tuple<>(),
146 args...
147 )
148 .construct_buffer_or_rebind();
149
150 KAMPING_ASSERT(internal::is_valid_rank_in_comm(source_param, *this, true, true));
151 int source = source_param.rank_signed();
152 int tag = tag_param.tag();
153 if constexpr (internal::has_to_be_computed<decltype(recv_count_param)>) {
154 Status probe_status = this->probe(source_param.clone(), tag_param.clone(), status_out()).extract_status();
155 source = probe_status.source_signed();
156 tag = probe_status.tag();
157 recv_count_param.underlying() = asserting_cast<int>(probe_status.count(recv_type.get_single_element()));
158 }
159
160 // Ensure that we do not touch the recv buffer if MPI_PROC_NULL is passed,
161 // because this is what the standard guarantees.
162 if constexpr (std::remove_reference_t<decltype(source_param)>::rank_type != internal::RankType::null) {
163 auto compute_required_recv_buf_size = [&] {
164 return asserting_cast<size_t>(recv_count_param.get_single_element());
165 };
166 recv_buf.resize_if_requested(compute_required_recv_buf_size);
167 KAMPING_ASSERT(
168 // if the recv type is user provided, kamping cannot make any assumptions about the required size of the
169 // recv buffer
170 recv_type_is_in_param || recv_buf.size() >= compute_required_recv_buf_size(),
171 "Recv buffer is not large enough to hold all received elements.",
172 assert::light
173 );
174 }
175
176 [[maybe_unused]] int err = MPI_Recv(
177 recv_buf.data(), // buf
178 recv_count_param.get_single_element(), // count
179 recv_type.get_single_element(), // datatype
180 source, // source
181 tag, // tag
182 this->mpi_communicator(), // comm
183 internal::status_param_to_native_ptr(status) // status
184 );
185 this->mpi_error_hook(err, "MPI_Recv");
186
187 return internal::make_mpi_result<std::tuple<Args...>>(
188 deserialization_repack<is_serialization_used>(std::move(recv_buf)),
189 std::move(recv_count_param),
190 std::move(status),
191 std::move(recv_type)
192 );
193}

195/// @brief Convience wrapper for receiving single values via \c MPI_Recv.
196///
197/// This wraps \c MPI_Recv. This operation performs a standard blocking receive with a receive count of 1 and returns
198/// the received value.
199///
200/// The following parameters are optional:
201/// - \ref kamping::source() receive a message sent from this source rank. Defaults to
202/// <code>kamping::source(kamping::rank::any)</code>.
203/// - \ref kamping::tag() recv message with this tag. Defaults to receiving for an arbitrary tag, i.e.
204/// <code>tag(tags::any)</code>.
205/// - \ref kamping::status() Returns info about the received message by setting the appropriate fields in the status
206/// object passed by the user. The status can be ignored by passing \c kamping::status(kamping::ignore<>). This is the
207/// default.
208///
209/// @tparam recv_value_type_tparam The type of the message to be received.
210/// @tparam Args Automatically deduced template parameters.
211/// @param args All required and any number of the optional buffers described above.
212/// @return The received value of type \c recv_value_type_tparam.
213template <
214 template <typename...>
215 typename DefaultContainerType,
216 template <typename, template <typename...> typename>
217 typename... Plugins>
218template <typename recv_value_type_tparam, typename... Args>
219auto kamping::Communicator<DefaultContainerType, Plugins...>::recv_single(Args... args) const {
220 KAMPING_CHECK_PARAMETERS(Args, KAMPING_REQUIRED_PARAMETERS(), KAMPING_OPTIONAL_PARAMETERS(tag, source, status));
221
222 using default_source_buf_type = decltype(kamping::source(rank::any));
223 using source_param_type = std::remove_reference_t<decltype(internal::select_parameter_type_or_default<
224 internal::ParameterType::source,
225 default_source_buf_type>({}, args...))>;
226 static_assert(
227 source_param_type::rank_type != internal::RankType::null,
228 "You cannot receive an element from source kamping::rank::null."
229 );
230
231 using default_status_param_type = decltype(kamping::status(kamping::ignore<>));
232 using status_param_type = std::remove_reference_t<decltype(internal::select_parameter_type_or_default<
233 internal::ParameterType::status,
234 default_status_param_type>({}, args...))>;
235 static_assert(
236 !internal::is_extractable<status_param_type>,
237 "KaMPIng cannot allocate a status object for you here, because we have no way of returning it. Pass a "
238 "reference to a status object instead."
239 );
240 return recv(recv_count(1), recv_buf(alloc_new<recv_value_type_tparam>), std::forward<Args>(args)...);
241}
