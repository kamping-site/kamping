29/// @brief Descriptor for different levels for message envelopes used in indirect communication.
30enum MessageEnvelopeLevel {
31 no_envelope, ///< do not use an envelope at all (if possible)
32 source, ///< only additionally add the source PE in the envelope (if possible)
33 source_and_destination ///< add source and destination PE in the envelope

76/// @brief Augments a plain message with additional information via \tparam Attributes
77/// @tparam PayloadType Plain message type.
78/// @tparam Attributes source or destination information
79template <typename PayloadType, typename... Attributes>
80struct MessageEnvelope : public Attributes... {
81 using Payload = PayloadType; ///< Underlying Type of message.
82 static constexpr bool has_source_information =
83 internal::type_list<Attributes...>::template contains<Source>; ///< Indicates whether the envelope contains the
84 ///< source PE.
85 static constexpr bool has_destination_information =
86 internal::type_list<Attributes...>::template contains<Destination>; ///< Indicates whether the envelope contains
87 ///< the destination PE.
88 /// @brief Default constructor.
89 MessageEnvelope() = default;
90
91 /// @brief Constructor for wrapping a message.
92 MessageEnvelope(PayloadType payload) : _payload{std::move(payload)} {}
93
94 /// @brief Return reference to payload.
95 Payload& get_payload() {
96 return _payload;
97 }
98
99 /// @brief Return const reference to payload.
100 Payload const& get_payload() const {
101 return _payload;
102 }
103
104 /// @brief Output operator.
105 friend std::ostream& operator<<(std::ostream& out, MessageEnvelope const& msg) {
106 out << "(payload: " << msg.get_payload();
107 if constexpr (MessageEnvelope::has_source_information) {
108 out << ", source: " << msg.get_source();
109 }
110 if constexpr (MessageEnvelope::has_destination_information) {
111 out << ", destination: " << msg.get_destination();
112 }
113 out << ")";
114 return out;
115 }
116
117 Payload _payload; ///< payload
118};

139/// @brief Object returned by \ref plugin::GridCommunicator::make_grid_communicator() representing a grid
140/// communicator which enables alltoall communication with a latency in `sqrt(p)` where p is the size of the
141/// original communicator.
142/// @tparam DefaultContainerType Container type of the original communicator.
143template <template <typename...> typename DefaultContainerType>
144class GridCommunicator {
145public:
146 using LevelCommunicator = kamping::Communicator<DefaultContainerType>; ///< Type of row and column communicator.
147
148 /// @brief Creates a two dimensional grid by splitting the given communicator of size `p` into a row and a column
149 /// communicator each of size about `sqrt(p)`.
150 /// @tparam Comm Type of the communicator.
151 /// @param comm Communicator to be split into a two-dimensional grid.
152 template <
153 template <typename...> typename = DefaultContainerType,
154 template <typename, template <typename...> typename>
155 typename... Plugins>
156 GridCommunicator(kamping::Communicator<DefaultContainerType, Plugins...> const& comm)
157 : _size_of_orig_comm{comm.size()},
158 _rank_in_orig_comm{comm.rank()} {
159 double const sqrt = std::sqrt(comm.size());
160 size_t const floor_sqrt = static_cast<size_t>(std::floor(sqrt));
161 size_t const ceil_sqrt = static_cast<size_t>(std::ceil(sqrt));
162 // We want to ensure that #columns + 1 >= #rows >= #columns.
163 // Therefore, use floor(sqrt(comm.size())) columns unless we have enough PEs to begin another row when using
164 // ceil(sqrt(comm.size()) columns.
165 size_t const threshold = floor_sqrt * ceil_sqrt;
166 _num_columns = (comm.size() >= threshold) ? ceil_sqrt : floor_sqrt;
167 size_t const num_ranks_in_incomplete_column = comm.size() / _num_columns;
168 auto [row, col] = pos_in_complete_grid(comm.rank()); // assume that we have a complete grid,
169 _size_complete_rectangle = _num_columns * num_ranks_in_incomplete_column;
170 if (comm.rank() >= _size_complete_rectangle) {
171 row = comm.rank() % _num_columns; // rank() is member of last incomplete row,
172 // therefore append it to one of the first
173 }
174 {
175 auto split_comm = comm.split(static_cast<int>(row), comm.rank_signed());
176 _row_comm = LevelCommunicator(split_comm.disown_mpi_communicator(), split_comm.root_signed(), true);
177 }
178 {
179 auto split_comm = comm.split(static_cast<int>(col), comm.rank_signed());
180 _column_comm = LevelCommunicator(split_comm.disown_mpi_communicator(), split_comm.root_signed(), true);
181 }
182 }

184 /// @brief Indirect two dimensional grid based personalized alltoall exchange.
185 /// The following parameters are required:
186 /// - \ref kamping::send_buf() containing the data that is sent to each rank. The size of this buffer has to be at
187 /// least the sum of the send_counts argument.
188 /// - \ref kamping::send_counts() containing the number of elements to send to each rank.
189 /// - \ref kamping::send_displs() containing the number of elements to send to each rank.
190 ///
191 /// The following parameters are optional:
192 /// - \ref kamping::send_displs() containing the offsets of the messages in send_buf. The `send_counts[i]` elements
193 /// starting at `send_buf[send_displs[i]]` will be sent to rank `i`. If omitted, this is calculated as the exclusive
194 /// prefix-sum of `send_counts`.
195 ///
196 /// Internally, each element in the send buffer is wrapped in an envelope to facilitate the indirect routing. The
197 /// envelope consists at least of the destination PE of each element but can be extended to also hold the
198 /// source PE of the element. The caller can specify whether they want to keep this information also in the output
199 /// via the \tparam envelope_level.
200 ///
201 /// @tparam envelope_level Determines the contents of the envelope of each returned element (no_envelope = use the
202 /// actual data type of an exchanged element, source = augment the actual data type with the source PE,
203 /// source_and_destination = argument the actual data type with the source and destination PE).
204 /// @tparam Args Automatically deducted template parameters.
205 /// @param args All required and any number of the optional buffers described above.
206 /// @returns
207 template <MessageEnvelopeLevel envelope_level = MessageEnvelopeLevel::no_envelope, typename... Args>
208 auto alltoallv_with_envelope(Args... args) const {
209 KAMPING_CHECK_PARAMETERS(
210 Args,
211 KAMPING_REQUIRED_PARAMETERS(send_buf, send_counts),
212 KAMPING_OPTIONAL_PARAMETERS(send_displs)
213 );
214 // Get send_buf
215 auto const& send_buf =
216 internal::select_parameter_type<internal::ParameterType::send_buf>(args...).construct_buffer_or_rebind();
217 // Get send_counts
218 auto const& send_counts = internal::select_parameter_type<internal::ParameterType::send_counts>(args...)
219 .template construct_buffer_or_rebind<DefaultContainerType>();
220
221 using default_send_displs_type = decltype(kamping::send_displs_out(alloc_new<DefaultContainerType<int>>));
222 auto&& send_displs =
223 internal::select_parameter_type_or_default<internal::ParameterType::send_displs, default_send_displs_type>(
224 std::tuple<>(),
225 args...
226 )
227 .template construct_buffer_or_rebind<DefaultContainerType>();
228 // Calculate send_displs if necessary
229 constexpr bool do_calculate_send_displs = internal::has_to_be_computed<decltype(send_displs)>;
230 if constexpr (do_calculate_send_displs) {
231 send_displs.resize_if_requested([&]() { return _size_of_orig_comm; });
232 std::exclusive_scan(send_counts.data(), send_counts.data() + _size_of_orig_comm, send_displs.data(), 0);
233 }
234 auto rowwise_recv_buf = rowwise_exchange<envelope_level>(send_buf, send_counts, send_displs);
235 return columnwise_exchange<envelope_level>(std::move(rowwise_recv_buf));
236 }

238 /// @brief Indirect two dimensional grid based personalized alltoall exchange.
239 /// The following parameters are required:
240 /// - \ref kamping::send_buf() containing the data that is sent to each rank. The size of this buffer has to be at
241 /// least the sum of the send_counts argument.
242 ///
243 /// - \ref kamping::send_counts() containing the number of elements to send to each rank.
244 ///
245 /// The following parameters are optional:
246 /// - \ref kamping::send_displs() containing the offsets of the messages in send_buf. The `send_counts[i]` elements
247 /// starting at `send_buf[send_displs[i]]` will be sent to rank `i`. If omitted, this is calculated as the exclusive
248 /// prefix-sum of `send_counts`.
249 ///
250 /// - \ref kamping::recv_counts() containing the number of elements to receive from each rank.
251 ///
252 /// - \ref kamping::recv_buf() containing a buffer for the output. Afterwards, this buffer will contain
253 ///
254 /// the data received as specified for send_buf. The buffer will be resized according to the buffer's
255 /// kamping::BufferResizePolicy. If resize policy is kamping::BufferResizePolicy::no_resize, the buffer's underlying
256 /// storage must be large enough to store all received elements.
257 ///
258 /// Internally, \ref alltoallv_with_envelope() is called.
259 ///
260 /// @tparam envelope_level Determines the contents of the envelope of each returned element (no_envelope = use the
261 /// actual data type of an exchanged element, source = augment the actual data type with the source PE,
262 /// source_and_destination = argument the actual data type with the source and destination PE).
263 /// @tparam Args Automatically deducted template parameters.
264 /// @param args All required and any number of the optional buffers described above.
265 /// @return Result type wrapping the output buffer and recv_counts (if requested).
266 template <typename... Args>
267 auto alltoallv(Args... args) const {
268 KAMPING_CHECK_PARAMETERS(
269 Args,
270 KAMPING_REQUIRED_PARAMETERS(send_buf, send_counts),
271 KAMPING_OPTIONAL_PARAMETERS(send_displs, recv_buf, recv_counts, recv_displs)
272 );
273 constexpr MessageEnvelopeLevel envelope_level = MessageEnvelopeLevel::source;
274 // get send_buf
275 auto& send_buf = internal::select_parameter_type<internal::ParameterType::send_buf>(args...);
276 using send_value_type = typename std::remove_reference_t<decltype(send_buf)>::value_type;
277 using default_recv_value_type = std::remove_const_t<send_value_type>;
278 // get send_counts
279 auto const& send_counts = internal::select_parameter_type<internal::ParameterType::send_counts>(args...)
280 .template construct_buffer_or_rebind<DefaultContainerType>();
281
282 using default_send_displs_type = decltype(kamping::send_displs_out(alloc_new<DefaultContainerType<int>>));
283 auto&& send_displs =
284 internal::select_parameter_type_or_default<internal::ParameterType::send_displs, default_send_displs_type>(
285 std::tuple<>(),
286 args...
287 )
288 .template construct_buffer_or_rebind<DefaultContainerType>();
289 // Calculate send_displs if necessary
290 constexpr bool do_calculate_send_displs = internal::has_to_be_computed<decltype(send_displs)>;
291 if constexpr (do_calculate_send_displs) {
292 send_displs.resize_if_requested([&]() { return _size_of_orig_comm; });
293 std::exclusive_scan(send_counts.data(), send_counts.data() + _size_of_orig_comm, send_displs.data(), 0);
294 }
295
296 // perform the actual message exchange
297 auto grid_recv_buf = alltoallv_with_envelope<envelope_level>(
298 std::move(send_buf),
299 kamping::send_counts(send_counts.underlying()),
300 kamping::send_displs(send_displs.underlying())
301 );
302
303 // post-processing (fixing ordering problems etc.)
304 // Get recv counts
305 using default_recv_counts_type = decltype(kamping::recv_counts_out(alloc_new<DefaultContainerType<int>>));
306 auto&& recv_counts =
307 internal::select_parameter_type_or_default<internal::ParameterType::recv_counts, default_recv_counts_type>(
308 std::tuple<>(),
309 args...
310 )
311 .template construct_buffer_or_rebind<DefaultContainerType>();
312 constexpr bool do_calculate_recv_counts = internal::has_to_be_computed<decltype(recv_counts)>;
313
314 if constexpr (do_calculate_recv_counts) {
315 recv_counts.resize_if_requested([&]() { return _size_of_orig_comm; });
316 KAMPING_ASSERT(
317 recv_counts.size() >= _size_of_orig_comm,
318 "Recv counts buffer is not large enough.",
319 assert::light
320 );
321 Span recv_counts_span(recv_counts.data(), recv_counts.size());
322 std::fill(recv_counts_span.begin(), recv_counts_span.end(), 0);
323
324 for (size_t i = 0; i < grid_recv_buf.size(); ++i) {
325 ++recv_counts_span[grid_recv_buf[i].get_source()];
326 }
327 } else {
328 KAMPING_ASSERT(
329 recv_counts.size() >= _size_of_orig_comm,
330 "Recv counts buffer is not large enough.",
331 assert::light
332 );
333 }
334
335 // Get recv displs
336 using default_recv_displs_type = decltype(kamping::recv_displs_out(alloc_new<DefaultContainerType<int>>));
337 auto&& recv_displs =
338 internal::select_parameter_type_or_default<internal::ParameterType::recv_displs, default_recv_displs_type>(
339 std::tuple<>(),
340 args...
341 )
342 .template construct_buffer_or_rebind<DefaultContainerType>();
343 constexpr bool do_calculate_recv_displs = internal::has_to_be_computed<decltype(recv_displs)>;
344
345 if constexpr (do_calculate_recv_displs) {
346 recv_displs.resize_if_requested([&]() { return _size_of_orig_comm; });
347 KAMPING_ASSERT(
348 recv_displs.size() >= _size_of_orig_comm,
349 "Recv displs buffer is not large enough.",
350 assert::light
351 );
352 Span recv_displs_span(recv_displs.data(), recv_displs.size());
353 Span recv_counts_span(recv_counts.data(), recv_counts.size());
354 std::exclusive_scan(recv_counts_span.begin(), recv_counts_span.end(), recv_displs_span.begin(), 0);
355 }
356
357 // get recv_buf
358 using default_recv_buf_type =
359 decltype(kamping::recv_buf(alloc_new<DefaultContainerType<default_recv_value_type>>));
360 auto&& recv_buf =
361 internal::select_parameter_type_or_default<internal::ParameterType::recv_buf, default_recv_buf_type>(
362 std::tuple<>(),
363 args...
364 )
365 .template construct_buffer_or_rebind<DefaultContainerType>();
366
367 write_recv_buffer(grid_recv_buf, recv_buf, recv_counts, recv_displs);
368
369 return internal::make_mpi_result<std::tuple<Args...>>(
370 std::move(send_displs),
371 std::move(recv_buf),
372 std::move(recv_counts),
373 std::move(recv_displs)
374 );
375 }

377private:
378 template <typename GridRecvBuffer, typename RecvBuffer, typename RecvCounts, typename RecvDispls>
379 void write_recv_buffer(
380 GridRecvBuffer const& grid_recv_buffer,
381 RecvBuffer& recv_buf,
382 RecvCounts const& recv_counts,
383 RecvDispls const& recv_displs
384 ) const {
385 auto write_pos = recv_displs.underlying();
386 Span write_pos_span(write_pos.data(), write_pos.size());
387 auto compute_required_recv_buf_size = [&]() {
388 Span recv_counts_span(recv_counts.data(), recv_counts.size());
389 return asserting_cast<size_t>(write_pos_span.back() + recv_counts_span.back());
390 };
391
392 recv_buf.resize_if_requested(compute_required_recv_buf_size);
393 KAMPING_ASSERT(
394 recv_buf.size() >= compute_required_recv_buf_size(),
395 "Recv buffer is not large enough to hold all received elements.",
396 assert::light
397 );
398
399 Span recv_buf_span(recv_buf.data(), recv_buf.size());
400 Span grid_recv_buf_span(grid_recv_buffer.data(), grid_recv_buffer.size());
401 for (auto const& elem: grid_recv_buf_span) {
402 size_t const pos = asserting_cast<size_t>(write_pos_span[elem.get_source()]++);
403 recv_buf_span[pos] = std::move(elem.get_payload());
404 }
405 }

407 template <typename SendCounts>
408 [[nodiscard]] auto compute_row_send_counts(SendCounts const& send_counts) const {
409 DefaultContainerType<int> row_send_counts(_row_comm.size(), 0);
410 for (size_t i = 0; i < send_counts.size(); ++i) {
411 size_t const destination_pe = get_destination_in_rowwise_exchange(i);
412 row_send_counts.data()[destination_pe] += send_counts.data()[i];
413 }
414 return row_send_counts;
415 }

417 [[nodiscard]] grid_plugin_helpers::GridPosition pos_in_complete_grid(size_t rank) const {
418 return grid_plugin_helpers::GridPosition{rank / _num_columns, rank % _num_columns};
419 }

421 [[nodiscard]] size_t get_destination_in_rowwise_exchange(size_t destination_rank) const {
422 return pos_in_complete_grid(destination_rank).col_index;
423 }

425 [[nodiscard]] size_t get_destination_in_colwise_exchange(size_t destination_rank) const {
426 return pos_in_complete_grid(destination_rank).row_index;
427 }

429 template <MessageEnvelopeLevel envelope_level, typename SendBuffer, typename SendCounts, typename SendDispls>
430 auto
431 rowwise_exchange(SendBuffer const& send_buf, SendCounts const& send_counts, SendDispls const& send_displs) const {
432 using namespace grid_plugin_helpers;
433 auto const row_send_counts = compute_row_send_counts(send_counts.underlying());
434 DefaultContainerType<int> row_send_displs(_row_comm.size());
435 Span row_send_counts_span(row_send_counts.data(), row_send_counts.size());
436 Span row_send_displs_span(row_send_displs.data(), row_send_displs.size());
437 std::exclusive_scan(
438 row_send_counts_span.begin(),
439 row_send_counts_span.end(),
440 row_send_displs_span.begin(),
441 int(0)
442 );
443 auto index_displacements = row_send_displs;
444 auto const total_send_count = static_cast<size_t>(row_send_displs_span.back() + row_send_counts_span.back());
445
446 using value_type = typename SendBuffer::value_type;
447 using MsgType = std::conditional_t<
448 envelope_level == MessageEnvelopeLevel::no_envelope,
449 MessageEnvelope<value_type, Destination>,
450 MessageEnvelope<value_type, Source, Destination>>;
451 DefaultContainerType<MsgType> rowwise_send_buf(total_send_count);
452
453 for (size_t i = 0; i < send_counts.size(); ++i) {
454 size_t const send_count = asserting_cast<size_t>(send_counts.underlying().data()[i]);
455 if (send_count == 0) {
456 continue;
457 }
458 int const destination = asserting_cast<int>(i);
459 auto const destination_in_row = get_destination_in_rowwise_exchange(asserting_cast<size_t>(i));
460 size_t const cur_displacement = asserting_cast<size_t>(send_displs.data()[i]);
461 for (std::size_t ii = 0; ii < send_count; ++ii) {
462 auto elem = send_buf.data()[cur_displacement + ii];
463 auto const idx = index_displacements[destination_in_row]++;
464 auto& entry = rowwise_send_buf[static_cast<size_t>(idx)];
465 entry = MsgType(std::move(elem));
466 entry.set_destination(destination
467 ); // this has to be done independently of the envelope level, otherwise routing is not possible
468 if constexpr (envelope_level != MessageEnvelopeLevel::no_envelope) {
469 entry.set_source(asserting_cast<int>(_rank_in_orig_comm));
470 }
471 }
472 }
473 return _row_comm.alltoallv(
474 kamping::send_buf(rowwise_send_buf),
475 kamping::send_counts(row_send_counts),
476 kamping::send_displs(row_send_displs)
477 );
478 }

480 template <
481 MessageEnvelopeLevel envelope_level,
482 template <typename...>
483 typename RowwiseRecvBuf,
484 typename... EnvelopeArgs>
485 auto columnwise_exchange(RowwiseRecvBuf<grid_plugin_helpers::MessageEnvelope<EnvelopeArgs...>>&& rowwise_recv_buf
486 ) const {
487 using namespace grid_plugin_helpers;
488 using IndirectMessageType = MessageEnvelope<EnvelopeArgs...>;
489 using Payload = typename IndirectMessageType::Payload;
490
491 DefaultContainerType<int> send_counts(_column_comm.size(), 0);
492 for (auto const& elem: rowwise_recv_buf) {
493 ++send_counts[get_destination_in_colwise_exchange(elem.get_destination())];
494 }
495 DefaultContainerType<int> send_offsets = send_counts;
496 Span send_counts_span(send_counts.data(), send_counts.size());
497 Span send_offsets_span(send_offsets.data(), send_offsets.size());
498 std::exclusive_scan(send_counts_span.begin(), send_counts_span.end(), send_offsets_span.begin(), int(0));
499 DefaultContainerType<int> send_displacements = send_offsets;
500
501 using MsgType = MessageEnvelopeType<envelope_level, Payload>;
502 DefaultContainerType<MsgType> colwise_send_buf(rowwise_recv_buf.size());
503 for (auto& elem: rowwise_recv_buf) {
504 auto const dst_in_column = get_destination_in_colwise_exchange(elem.get_destination());
505 auto& entry = colwise_send_buf[static_cast<std::size_t>(send_offsets[dst_in_column]++)];
506 entry = MsgType(std::move(elem.get_payload()));
507 if constexpr (envelope_level == MessageEnvelopeLevel::no_envelope) {
508 // nothing to be done
509 } else if constexpr (envelope_level == MessageEnvelopeLevel::source) {
510 entry.set_source(elem.get_source_signed());
511 } else if constexpr (envelope_level == MessageEnvelopeLevel::source_and_destination) {
512 entry.set_source(elem.get_source_signed());
513 entry.set_destination(elem.get_destination_signed());
514 } else {
515 static_assert(
516 envelope_level == MessageEnvelopeLevel::source_and_destination /*always false*/,
517 "This branch is never used"
518 );
519 }
520 }
521 {
522 // deallocate rowwise_recv_buf as it is not needed anymore
523 rowwise_recv_buf.clear();
524 auto tmp = std::move(rowwise_recv_buf);
525 }
526 return _column_comm.alltoallv(
527 kamping::send_buf(colwise_send_buf),
528 kamping::send_counts(send_counts),
529 kamping::send_displs(send_displacements)
530 );
531 }
532
533private:
534 size_t _size_of_orig_comm;
535 size_t _rank_in_orig_comm;
536 size_t _size_complete_rectangle;
537 size_t _num_columns;
538 kamping::Communicator<DefaultContainerType> _row_comm;
539 kamping::Communicator<DefaultContainerType> _column_comm;
540};

543/// @brief Plugin adding a two dimensional communication grid to the communicator.
544///
545/// PEs are row-major and abs(`#row - #columns`) <= 1
546/// 0 1 2 3
547/// 4 5 6 7
548/// 8 9 10 11
549/// 12 13 14 15
550///
551/// If `#PE != #row * #column` then the PEs of the last incomplete row are transposed and appended to
552/// the first rows and do not form an own row based communicator.
553/// 0 1 2 3 16
554/// 4 5 6 7 17
555/// 8 9 10 11
556/// 12 13 14 15
557/// (16 17)
558/// This enables personalized alltoall exchanges with a latency in about `sqrt(#PE)`.
559template <typename Comm, template <typename...> typename DefaultContainerType>
560class GridCommunicator : public plugin::PluginBase<Comm, DefaultContainerType, GridCommunicator> {
561public:

563 auto make_grid_communicator() const {
564 return grid::GridCommunicator<DefaultContainerType>(this->to_communicator());
565 }
566};
