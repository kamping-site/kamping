22/// @brief Class representing a node in the timer tree. Each node represents a time measurement (or multiple with
23/// the
24/// same key). A node can have multiple children which represent nested time measurements. The measurements
25/// associated with a node's children are executed while the node's measurement is still active.
26///
27/// @tparam Duration Type of a duration.
28template <typename Duration>
29class AggregatedTreeNode : public internal::TreeNode<AggregatedTreeNode<Duration>> {
30public:
31 using internal::TreeNode<AggregatedTreeNode<Duration>>::TreeNode;
32
33 ///@brief Type into which the aggregated data is stored together with the applied aggregation operation.
34 using StorageType = std::unordered_map<GlobalAggregationMode, std::vector<ScalarOrContainer<Duration>>>;
35
36 /// @brief Access to stored aggregated data.
37 /// @return Reference to aggregated data.
38 auto const& aggregated_data() const {
39 return _aggregated_data;
40 }
41
42 /// @brief Add scalar of type T to aggregated data storage together with the name of the applied aggregation
43 /// operation.
44 /// @param aggregation_mode Aggregation mode that has been applied to the duration data.
45 /// @param data Scalar resulted from applying the given aggregation operation.
46 void add(GlobalAggregationMode aggregation_mode, std::optional<Duration> data) {
47 if (data) {
48 _aggregated_data[aggregation_mode].emplace_back(data.value());
49 }
50 }
51
52 /// @brief Add scalar of type T to aggregated data storage together with the name of the applied aggregation
53 /// operation.
54 /// @param aggregation_mode Aggregation mode that has been applied to the duration data.
55 /// @param data Vector of Scalars resulted from applying the given aggregation operation.
56 void add(GlobalAggregationMode aggregation_mode, std::vector<Duration> const& data) {
57 _aggregated_data[aggregation_mode].emplace_back(data);
58 }
59
60public:
61 StorageType _aggregated_data; ///< Storage of the aggregated data.
62};
63} // namespace kamping::measurements
